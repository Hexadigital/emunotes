0xC0BE - Player Direction
	0 = Down
	1 = Right
	2 = Up
	3 = Left
	4 = Down, Walking
	5 = Right, Walking
	6 = Up, Walking
	7 = Left, Walking
	8 = Full screen glitch (directionless)

	1 = Obtained
